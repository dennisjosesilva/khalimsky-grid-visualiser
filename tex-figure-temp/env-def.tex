\usepackage[framemethod=TikZ]{mdframed}

\definecolor{definitioncolour}{RGB}{0,102,0}
\definecolor{propositioncolour}{RGB}{0,0,102}

\newcounter{definition}[chapter]
\newenvironment{definition}[1]{%
	\refstepcounter{definition}%
	\mdfsetup{%
		frametitle={%
			\tikz[baseline=(current bounding box.east), outer sep=0pt]
			\node[anchor=east, rectangle, fill=definitioncolour!40]
			{\strut Definition~\thedefinition};
		}%	
	}%
	\mdfsetup{innertopmargin=10pt,linecolor=definitioncolour!40,linewidth=2pt,%
	topline=true,frametitleaboveskip=\dimexpr-\ht\strutbox\relax}
	\begin{mdframed}[]\relax%
	\ifstrempty{#1}{}{ \textbf{#1:} } }
	{\end{mdframed}}
	
% proposition
\newcounter{proposition}[chapter] \setcounter{proposition}{0}
\renewcommand{\theproposition}{\arabic{chapter}.\arabic{proposition}}
\newenvironment{proposition}{%
	\refstepcounter{proposition}%
	\mdfsetup{%
		frametitle={%
			\tikz[baseline=(current bounding box.east), outer sep=0pt]
			\node[anchor=east, rectangle, fill=propositioncolour!40]
			{\strut Proposition~\theproposition};
		}%
	}%
	\mdfsetup{innertopmargin=10pt,linecolor=propositioncolour!40,linewidth=2pt,%
	topline=true,frametitleaboveskip=\dimexpr-\ht\strutbox\relax}
	\begin{mdframed}[]\relax}
	{\end{mdframed}}

%Theorem
\newcounter{theo}[chapter] \setcounter{theo}{0}
\renewcommand{\thetheo}{\arabic{chapter}.\arabic{theo}}
\newenvironment{theo}[2][]{%
\refstepcounter{theo}%
\ifstrempty{#1}%
{\mdfsetup{%
frametitle={%
\tikz[baseline=(current bounding box.east),outer sep=0pt]
\node[anchor=east,rectangle,fill=blue!20]
{\strut Theorem~\thetheo};}}
}%
{\mdfsetup{%
frametitle={%
\tikz[baseline=(current bounding box.east),outer sep=0pt]
\node[anchor=east,rectangle,fill=blue!20]
{\strut Theorem~\thetheo:~#1};}}%
}%
\mdfsetup{innertopmargin=10pt,linecolor=blue!20,%
linewidth=2pt,topline=true,%
frametitleaboveskip=\dimexpr-\ht\strutbox\relax
}
\begin{mdframed}[]\relax%
\label{#2}}{\end{mdframed}}
